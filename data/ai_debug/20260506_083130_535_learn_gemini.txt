def parse(text: str) -> list:
    rows = []

    # Split the text into blocks, where each block corresponds to a Bill of Lading.
    # A new BL block is identified by the shipper information, which starts with "1) "
    # followed by a name (usually starting with an uppercase letter).
    bl_blocks = re.split(r'\n(?=1\)\s+[A-Z])', text)

    # Extract global header info as fallbacks
    pol_header_match = re.search(r'5a\.Port of loading\s+([^\n]+)', text)
    pol_fallback = pol_header_match.group(1).strip() if pol_header_match else None
    pod_header_match = re.search(r'5b\.Port of discharge\s+([^\n]+)', text)
    pod_fallback = pod_header_match.group(1).strip() if pod_header_match else None

    for block in bl_blocks:
        bl_match = re.search(r'\b(ABJ\d{9,10})\b', block)
        if not bl_match:
            continue
        bl_number = bl_match.group(1)

        shipper_match = re.search(r'^1\)\s+(.*?)(?=\n\s*2\))', block, re.DOTALL)
        shipper = ' '.join(shipper_match.group(1).strip().split()) if shipper_match else None

        consignee_match = re.search(r'\n2\)\s+(.*?)(?=\n\s*3\))', block, re.DOTALL)
        consignee = ' '.join(consignee_match.group(1).strip().split()) if consignee_match else None

        notify_match = re.search(r'\n3\)\s+(.*?)(?=\bABJ\d{9,10}|\(MN\)|B/L\. NO\.)', block, re.DOTALL)
        notify = ' '.join(notify_match.group(1).strip().split()) if notify_match else None

        weight_match = re.search(r'1\)\s+([\d,.]+)\s*KGS', block)
        weight = weight_match.group(1).replace(',', '') if weight_match else None
        if not weight:
            weight_match_alt = re.search(r'TOTAL GROSS WEIGHT:\s*([\d,.]+)\s*KGS|GROSS WEIGHT\s*[:=]\s*([\d,.\s]+)\s*KGS', block, re.IGNORECASE)
            if weight_match_alt:
                weight = (weight_match_alt.group(1) or weight_match_alt.group(2)).replace(',', '').strip()

        volume_match = re.search(r'2\)\s+([\d,.]+)\s*CBM', block)
        volume = volume_match.group(1).replace(',', '') if volume_match else None

        pack_qty_match = re.search(r'([\d,]+)\s+(BALE|BAGS?|CARTONS?)', block)
        pack_qty = pack_qty_match.group(1).replace(',', '') if pack_qty_match else None
        pack_unit = pack_qty_match.group(2) if pack_qty_match else None

        desc_match = re.search(r'SAID TO CONTAIN:\s*(.*?)(?=NET WEIGHT|GROSS WEIGHT|HS CODE|FREIGHT|DEC NO|D6 E)', block, re.DOTALL | re.IGNORECASE)
        description = ' '.join(desc_match.group(1).strip().split()) if desc_match else None

        pol_match = re.search(r'5a\.Port of loading\s+([^\n]+)', block)
        port_of_loading = pol_match.group(1).strip() if pol_match else pol_fallback

        pod_match = re.search(r'5b\.Port of discharge\s+([^\n]+)', block)
        port_of_discharge = pod_match.group(1).strip() if pod_match else pod_fallback

        delivery_match = re.search(r'Final destination\s+([^\n]+)', block)
        place_of_delivery = delivery_match.group(1).strip() if delivery_match else None

        container_matches = list(re.finditer(r'\(\s*CN\s*\)\s+([A-Z]{4}\d{7})', block))
        if not container_matches:
            rows.append({
                'bl_number': bl_number, 'shipper': shipper, 'consignee': consignee, 'notify': notify,
                'weight': weight, 'volume': volume, 'pack_qty': pack_qty, 'pack_unit': pack_unit,
                'description': description, 'port_of_loading': port_of_loading,
                'port_of_discharge': port_of_discharge, 'place_of_delivery': place_of_delivery,
            })
            continue

        for i, match in enumerate(container_matches):
            container_number = match.group(1)
            
            start_pos = match.start()
            end_pos = container_matches[i+1].start() if i + 1 < len(container_matches) else len(block)
            sub_block = block[start_pos:end_pos]

            seal_match = re.search(r'\(\s*SN\s*\)\s+([^\n/]+)', sub_block)
            seal1 = seal_match.group(1).strip() if seal_match else None

            ctype_match = re.search(r'\(\s*CY/CY\s*\)\s*-\s*(\w+)', sub_block)
            container_type = ctype_match.group(1).strip() if ctype_match else None

            rows.append({
                'bl_number': bl_number, 'shipper': shipper, 'consignee': consignee, 'notify': notify,
                'weight': weight, 'volume': volume, 'pack_qty': pack_qty, 'pack_unit': pack_unit,
                'description': description, 'port_of_loading': port_of_loading,
                'port_of_discharge': port_of_discharge, 'place_of_delivery': place_of_delivery,
                'container_number': container_number, 'container_type': container_type, 'seal1': seal1,
            })
            
    return rows
