def parse(text: str) -> list:
    rows = []

    # Split text into BL blocks. Using a lookahead to keep the delimiter.
    bl_blocks = re.split(r'(?=\nSH:)', text)
    if not bl_blocks:
        return []

    # The first block is the header, so we start from the first actual BL block.
    start_index = 0
    if not bl_blocks[0].strip().startswith('SH:'):
        start_index = 1

    for block in bl_blocks[start_index:]:
        block = block.strip()
        if not block:
            continue

        # Extract BL-level information
        bl_number_match = re.search(r'SH:\s*(\S+)', block)
        bl_number = bl_number_match.group(1) if bl_number_match else None

        shipper_match = re.search(r'SH:(.*?)(?=\nCO:)', block, re.DOTALL)
        shipper = ""
        if shipper_match:
            shipper_text = shipper_match.group(1).strip()
            # Remove the first line which contains the BL number
            shipper_text = re.sub(r'^\S+\s*', '', shipper_text, count=1).strip()
            shipper = ' '.join(shipper_text.split('\n')).strip()

        consignee_match = re.search(r'CO:(.*?)(?=\nNF:)', block, re.DOTALL)
        consignee = ' '.join(consignee_match.group(1).strip().split('\n')).strip() if consignee_match else ""

        notify_match = re.search(r'NF:(.*?)(?=\nEW:|\nTotal No. of containers:)', block, re.DOTALL)
        notify = ' '.join(notify_match.group(1).strip().split('\n')).strip() if notify_match else ""

        # Find all container sections within the block
        cn_iter = list(re.finditer(r'CN:\s*([A-Z]{4}\d{7})', block))
        
        if not cn_iter:
            continue

        for i, cm in enumerate(cn_iter):
            container_number = cm.group(1)
            
            cn_start = cm.start()
            cn_end = cn_iter[i+1].start() if i+1 < len(cn_iter) else len(block)
            cn_block = block[cn_start:cn_end]

            container_type_match = re.search(r'CN:\s*[A-Z]{4}\d{7}\s+(.*?)(?:\n| EXPRESS RELEASE)', cn_block)
            container_type = container_type_match.group(1).strip() if container_type_match else None

            seal_match = re.search(r'SN:\s*(\S+)', cn_block)
            seal1 = seal_match.group(1) if seal_match else None

            weight_match = re.search(r'(\d[\d,.]*)\s*kgs', cn_block)
            weight = weight_match.group(1).replace(',', '') if weight_match else None

            pack_qty = None
            pack_unit = None
            description = ""

            desc_start_match = re.search(r'\d+\s+[A-Z]+\(S\)', cn_block)
            if desc_start_match:
                desc_start_pos = desc_start_match.start()
                
                desc_end_match = re.search(r'Harmonized Code:', cn_block, flags=re.IGNORECASE)
                desc_end_pos = desc_end_match.start() if desc_end_match and desc_end_match.start() > desc_start_pos else len(cn_block)
                
                desc_text = cn_block[desc_start_pos:desc_end_pos]
                
                pack_match = re.search(r'^(\d+)\s+([A-Z]+\(S\))', desc_text)
                if pack_match:
                    pack_qty = pack_match.group(1)
                    pack_unit = pack_match.group(2)
                
                # Clean the description text by removing merged weight columns and package info
                desc_text_cleaned = re.sub(r'\s+\d[\d,.]*\s*(kgs|lbs)\.?', '', desc_text, flags=re.IGNORECASE)
                desc_text_cleaned = re.sub(r'^\d+\s+[A-Z]+\(S\)\s*', '', desc_text_cleaned)
                description = ' '.join(desc_text_cleaned.strip().split('\n')).strip()

            row = {
                'bl_number': bl_number,
                'container_number': container_number,
                'container_type': container_type,
                'seal1': seal1,
                'weight': weight,
                'weight_unit': 'KGS' if weight else None,
                'pack_qty': pack_qty,
                'pack_unit': pack_unit,
                'description': description,
                'shipper': shipper,
                'consignee': consignee,
                'notify': notify,
            }
            rows.append(row)

    return rows
