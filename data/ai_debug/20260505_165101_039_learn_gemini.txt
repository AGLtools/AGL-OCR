def parse(text: str) -> list:
    rows = []
    
    # Split the document into BL blocks. Each block starts with the shipper info.
    # The pattern `\n1)\s` is a reliable delimiter for a new shipper/BL section.
    blocks = re.split(r'\n1\)\s', text)
    if len(blocks) < 2:
        return []

    for block in blocks[1:]:
        # Prepend the "1) " we split on to make parsing consistent.
        current_block = "1) " + block

        bl_match = re.search(r'\b(ABJ\d{9,10})\b', current_block)
        if not bl_match:
            continue

        base_data = {'bl_number': bl_match.group(1)}
        
        # Extract parties
        shipper_match = re.search(r'^1\)\s*(.*?)(?=\n\s*2\)\s)', current_block, re.DOTALL)
        if shipper_match:
            base_data['shipper'] = ' '.join(shipper_match.group(1).strip().split())
        
        consignee_match = re.search(r'2\)\s*(.*?)(?=\n\s*3\)\s)', current_block, re.DOTALL)
        if consignee_match:
            base_data['consignee'] = ' '.join(consignee_match.group(1).strip().split())
        
        notify_match = re.search(r'3\)\s*(.*?)(?=\bABJ\d{9,10}\b|B/L\. NO\.)', current_block, re.DOTALL)
        if notify_match:
            base_data['notify'] = ' '.join(notify_match.group(1).strip().split())

        # Extract ports
        pol_match = re.search(r'5a\.Port of loading\s+([^\n]+)', current_block)
        if pol_match:
            base_data['port_of_loading'] = pol_match.group(1).strip()

        pod_match = re.search(r'5b\.Port of discharge\s+([^\n]+)', current_block)
        if pod_match:
            base_data['port_of_discharge'] = pod_match.group(1).strip()

        pld_match = re.search(r'Final destination\s+([^\n]+)', current_block)
        if pld_match:
            base_data['place_of_delivery'] = pld_match.group(1).strip()

        # Extract common cargo details
        weight_match = re.search(r'\n1\)\s*([\d,]+\.\d+)\s*KGS', current_block)
        if weight_match:
            base_data['weight'] = weight_match.group(1).replace(',', '')
            base_data['weight_unit'] = 'KGS'

        volume_match = re.search(r'\n2\)\s*([\d,]+\.\d+)\s*CBM', current_block)
        if volume_match:
            base_data['volume'] = volume_match.group(1).replace(',', '')
            base_data['volume_unit'] = 'CBM'

        pack_match = re.search(r'(\d[\d,\s]*)\s+(BALE|BAG|CARTON)S?', current_block, re.IGNORECASE)
        if pack_match:
            base_data['pack_qty'] = re.sub(r'[\s,]', '', pack_match.group(1))
            base_data['pack_unit'] = pack_match.group(2).upper()
            if base_data['pack_unit'] == 'CARTON':
                base_data['pack_unit'] = 'CARTONS'

        # Find all containers for this BL
        cn_iter = list(re.finditer(r'\(CN\)\s*([A-Z]{4}\d{7})', current_block))
        
        # Extract description
        last_container_line_end = 0
        if cn_iter:
            last_cn_match = cn_iter[-1]
            end_of_line_match = re.search(r'\n', current_block[last_cn_match.end():])
            if end_of_line_match:
                last_container_line_end = last_cn_match.end() + end_of_line_match.start()
        else:
            bl_line_end_match = re.search(r'\n', current_block[bl_match.end():])
            if bl_line_end_match:
                last_container_line_end = bl_match.end() + bl_line_end_match.start()

        weight_section_start = len(current_block)
        if weight_match:
            weight_section_start = weight_match.start()

        desc_text = current_block[last_container_line_end:weight_section_start]
        desc_text = re.sub(r'\(CN\).*?\n', '', desc_text)
        desc_text = re.sub(r'\(SN\).*?\n', '', desc_text)
        desc_text = re.sub(r'\(CY/CY\).*?\n', '', desc_text)
        base_data['description'] = ' '.join(desc_text.strip().split())

        if cn_iter:
            for i, cn_match in enumerate(cn_iter):
                row = base_data.copy()
                row['container_number'] = cn_match.group(1)
                
                start_pos = cn_match.start()
                end_pos = cn_iter[i+1].start() if i + 1 < len(cn_iter) else len(current_block)
                container_sub_block = current_block[start_pos:end_pos]
                
                seal_match = re.search(r'\(SN\)\s*([^\n]+)', container_sub_block)
                if seal_match:
                    row['seal1'] = seal_match.group(1).strip().replace('/', ' ')

                type_match = re.search(r'\(CY/CY\)\s*-\s*([A-Z0-9]+)', container_sub_block)
                if type_match:
                    row['container_type'] = type_match.group(1).strip()
                
                rows.append(row)
        else:
            rows.append(base_data)
            
    return rows
<<<END_PYTHON>>>
