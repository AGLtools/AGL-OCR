import re

def parse(text: str) -> list:
    rows = []

    # Regex pour extraire les informations d'en-tête de BL
    bl_header_pattern = re.compile(
        r"SH:\s*(?P<shipper>.*?)\s*CO:\s*(?P<consignee>.*?)\s*NF:\s*(?P<notify>.*?)(?=\n\s*CN:|Total No\. of containers:)",
        re.DOTALL
    )

    # Regex pour extraire les informations de conteneur
    container_pattern = re.compile(
        r"CN:(?P<container_number>[A-Z]{4}\d{7})\s*(?P<container_type>[0-9]{2}'\s*[A-Z\s]+(?:HIGH CUBE|DRY VAN)?)\s*(?P<bl_type>.*?)\s*SN:\s*(?P<sn>\d+)\s*Tare Weight:\s*(?P<tare_weight>[\d,\.]+)\s*kgs\.\s*(?P<pack_qty>\d+)\s*(?P<pack_unit>[A-Z\s\(\)]+)\s*of\s*(?P<description>.*?)\s*(?P<weight>[\d,\.]+)\s*kgs\.\s*(?P<weight_lbs>[\d,\.]+lbs\.)",
        re.DOTALL
    )

    # Regex pour le numéro de BL (peut être avant SH: ou dans le bloc SH/CO/NF)
    bl_number_pattern = re.compile(r"B/L NR\.\s*(?P<bl_number>[A-Z]{5}\d{7,})")
    
    # Regex pour le port de chargement et déchargement
    port_pattern = re.compile(
        r"5a\. PORT OF LOADING\s+5b\. PORT OF DISCHARGE\s+FINAL DESTINATION\s+Transshipment:\s*(?P<transshipment>.*?)\n"
        r"(?P<pol>[A-Z\s,]+)\s+(?P<pod>[A-Z\s,0-9]+)\s+\(not required in U\.S\.\)",
        re.DOTALL
    )

    # Extraire les ports une seule fois par page ou document si stable
    pol = ""
    pod = ""
    transshipment = ""
    port_match = port_pattern.search(text)
    if port_match:
        pol = port_match.group("pol").strip()
        pod = port_match.group("pod").strip()
        transshipment = port_match.group("transshipment").strip()

    # Diviser le texte en blocs de BL potentiels
    # Un nouveau BL commence par "SH:" ou "CN:" si le BL précédent n'avait pas de SH/CO/NF
    bl_blocks = re.split(r"(?=\nSH:|\nCN:[A-Z]{4}\d{7})", text)

    current_bl_data = {}

    for block in bl_blocks:
        # Tenter d'extraire le numéro de BL s'il est présent
        bl_num_match = bl_number_pattern.search(block)
        if bl_num_match:
            current_bl_data['bl_number'] = bl_num_match.group('bl_number').strip()

        # Tenter d'extraire les infos SH/CO/NF
        bl_header_match = bl_header_pattern.search(block)
        if bl_header_match:
            current_bl_data['shipper'] = bl_header_match.group('shipper').strip()
            current_bl_data['consignee'] = bl_header_match.group('consignee').strip()
            current_bl_data['notify'] = bl_header_match.group('notify').strip()
            # Nettoyer les adresses pour enlever les numéros de téléphone/fax et "EW:"
            current_bl_data['shipper'] = re.sub(r"Phone No :.*|Fax No :.*", "", current_bl_data['shipper'], flags=re.DOTALL).strip()
            current_bl_data['consignee'] = re.sub(r"Phone No :.*|Fax No :.*", "", current_bl_data['consignee'], flags=re.DOTALL).strip()
            current_bl_data['notify'] = re.sub(r"Phone No :.*|Fax No :.*|EW:", "", current_bl_data['notify'], flags=re.DOTALL).strip()

            # Si un nouveau BL est trouvé, réinitialiser les données spécifiques aux conteneurs
            # et s'assurer que les ports sont mis à jour
            current_bl_data['port_of_loading'] = pol
            current_bl_data['port_of_discharge'] = pod
            current_bl_data['place_of_delivery'] = transshipment # Utiliser transshipment comme place_of_delivery

        # Extraire les informations de conteneur
        for container_match in container_pattern.finditer(block):
            container_info = current_bl_data.copy() # Copier les infos du BL courant

            container_info['container_number'] = container_match.group('container_number').strip()
            container_info['container_type'] = container_match.group('container_type').strip()
            container_info['bl_type'] = container_match.group('bl_type').strip()
            container_info['pack_qty'] = int(float(container_match.group('pack_qty').replace(',', ''))) if container_match.group('pack_qty') else None
            container_info['pack_unit'] = container_match.group('pack_unit').strip()
            container_info['description'] = container_match.group('description').strip()
            container_info['weight'] = float(container_match.group('weight').replace(',', '')) if container_match.group('weight') else None
            container_info['weight_unit'] = 'KGS'

            # Extraire le numéro de page
            page_match = re.search(r"Page (\d+) of \d+", block)
            if page_match:
                container_info['page'] = int(page_match.group(1))

            rows.append(container_info)

    return rows
