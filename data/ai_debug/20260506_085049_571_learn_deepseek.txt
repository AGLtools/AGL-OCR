        return rows

    # To capture the first BL's party info (before any BL), we need to treat the beginning as block for first BL
    # We'll iterate over positions and define blocks as: from start of this BL to start of next BL
    for idx, (start, bl_number) in enumerate(bl_positions):
        # Block content: from the line containing this BL number to the line before the next BL number
        next_start = bl_positions[idx + 1][0] if idx + 1 < len(bl_positions) else len(text)
        # To include party info, we need to go back to the previous block's start? No, party info is usually before the BL number.
        # Better: take the block from the previous BL's party area to this BL's end? Actually, the party info for a BL appears BEFORE the BL number line.
        # So we need to capture the text from the end of the previous BL's content (or document start) up to this BL's start.
        if idx == 0:
            block = text[:next_start]
        else:
            # Start from the end of the previous BL's description area, but simpler: start from the line of the previous BL's number +1?
            # We'll use the current start - maybe 200 chars back to catch the party info
            prev_end = bl_positions[idx - 1][0] + len(bl_number)  # approximate
            block = text[max(0, prev_end - 200):next_start]  # crude but should work for this format

        # Extract shipper, consignee, notify from the block
        shipper = None
        consignee = None
        notify = None

        # Find the section with numbered parties
        # Pattern: "1) SHIPPER" then content until "2) CONSIGNEE"
        shipper_match = re.search(r'1\)\s*SHIPPER\s*\n(.*?)(?=\n\s*2\)\s*CONSIGNEE)', block, re.DOTALL | re.IGNORECASE)
