def parse(text: str) -> list:
    rows = []
    
    # Split the text into blocks, where each block represents a BL and starts with the shipper info.
    # A shipper block starts with "1) " but not "1) SHIPPER".
    # We use a positive lookahead on a newline to ensure we split by lines.
    bl_blocks = re.split(r'\n(?=1\)\s(?!SHIPPER))', text)

    for block in bl_blocks:
        bl_match = re.search(r'\b(ABJ\d{8,10})\b', block)
        if not bl_match:
            continue
        
        bl_number = bl_match.group(1)

        shipper_match = re.search(r'1\)\s*([\s\S]+?)\s*2\)', block)
        shipper = shipper_match.group(1).strip() if shipper_match else None

        consignee_match = re.search(r'2\)\s*([\s\S]+?)\s*3\)', block)
        consignee = consignee_match.group(1).strip() if consignee_match else None

        notify_match = re.search(r'3\)\s*([\s\S]+?)(?:\bABJ\d{8,10}\b|\(MN\)|B/L\. NO\.)', block)
        notify = notify_match.group(1).strip() if notify_match else None

        pol_match = re.search(r'5a\.Port of loading\s+([^\n]+)', block)
        port_of_loading = pol_match.group(1).strip() if pol_match else None

        pod_match = re.search(r'5b\.Port of discharge\s+([^\n]+)', block)
        port_of_discharge = pod_match.group(1).strip() if pod_match else None

        delivery_match = re.search(r'Final destination\s+([^\n]+)', block)
        place_of_delivery = delivery_match.group(1).strip() if delivery_match else None

        desc_match = re.search(r'CONTAINER\(S\)\s+SAID\s+TO\s+CONTAIN:\s*([\s\S]+?)(?:NET\s+WEIGHT|GROSS\s+WEIGH|FREIGHT\s+PAYABLE|SERVICE\s+CONTRACT|Colis:|HS\s+CODE)', block)
        description = desc_match.group(1).strip().replace('\n', ' ') if desc_match else ''

        pack_qty = None
        pack_unit = None
        pack_match = re.search(r'([\d,\s]+)\s*(BALE|BAG|CARTON)S?', block)
        if pack_match:
            try:
                pack_qty = int(re.sub(r'[\s,]', '', pack_match.group(1)))
                pack_unit = pack_match.group(2)
                if not pack_unit.endswith('S'):
                    pack_unit += 'S'
            except (ValueError, IndexError):
                pass

        weight = None
        weight_match = re.search(r'1\)\s*([\d,.]+)\s*KGS', block)
        if weight_match:
            try:
                weight = float(weight_match.group(1).replace(',', ''))
            except ValueError:
                pass

        volume = None
        volume_match = re.search(r'2\)\s*([\d,.]+)\s*CBM', block)
        if volume_match:
            try:
                volume = float(volume_match.group(1).replace(',', ''))
            except ValueError:
                pass

        cn_iter = list(re.finditer(r'\(CN\)\s+([A-Z]{4}\d{7})', block))
        if not cn_iter:
            continue

        for i, cn_match in enumerate(cn_iter):
            start_pos = cn_match.start()
            end_pos = cn_iter[i + 1].start() if i + 1 < len(cn_iter) else len(block)
            cn_sub_block = block[start_pos:end_pos]

            container_number = cn_match.group(1)
            
            seal_match = re.search(r'\(SN\)\s+([^\s/]+)', cn_sub_block)
            seal1 = seal_match.group(1) if seal_match else None
            
            ctype_match = re.search(r'\(CY/CY\)\s*-\s*([A-Z0-9]+)', cn_sub_block)
            container_type = ctype_match.group(1) if ctype_match else None

            row = {
                'bl_number': bl_number,
                'shipper': shipper,
                'consignee': consignee,
                'notify': notify,
                'port_of_loading': port_of_loading,
                'port_of_discharge': port_of_discharge,
                'place_of_delivery': place_of_delivery,
                'description': description,
                'pack_qty': pack_qty,
                'pack_unit': pack_unit,
                'weight': weight,
                'weight_unit': 'KGS' if weight is not None else None,
                'volume': volume,
                'volume_unit': 'CBM' if volume is not None else None,
                'container_number': container_number,
                'container_type': container_type,
                'seal1': seal1,
            }
            rows.append(row)
            
    return rows
