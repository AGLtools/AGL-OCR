def parse(text: str) -> list:
    rows = []

    def clean_text(value):
        if not value:
            return ""
        return ' '.join(value.strip().replace('\n', ' ').split())

    # Global header fields
    pol_match = re.search(r'5a\.Port of loading\s+([^\n]+)', text)
    port_of_loading = clean_text(pol_match.group(1)) if pol_match else None

    pod_match = re.search(r'5b\.Port of discharge\s+([^\n]+)', text)
    port_of_discharge = clean_text(pod_match.group(1)) if pod_match else None

    # Find all B/L numbers and potential party block starts
    bl_matches = list(re.finditer(r'\b(ABJ\d{8,11})\b', text))
    party_starts = list(re.finditer(r'\n1\)\s+[A-Z]', text))

    if not bl_matches:
        return []

    for i, bl_match in enumerate(bl_matches):
        bl_number = bl_match.group(1)

        # Find the corresponding party block start for this B/L
        # It's the last party_start marker that appears before the current B/L number
        current_party_start_pos = 0
        for ps in party_starts:
            if ps.start() < bl_match.start():
                current_party_start_pos = ps.start()
            else:
                break
        
        # Define party and content blocks based on B/L number positions
        party_block = text[current_party_start_pos:bl_match.start()]
        content_start = bl_match.end()
        content_end = bl_matches[i+1].start() if i + 1 < len(bl_matches) else len(text)
        content_block = text[content_start:content_end]

        # Extract from party_block
        shipper_match = re.search(r'1\)\s*([\s\S]+?)(?=\n\s*2\))', party_block)
        shipper = clean_text(shipper_match.group(1)) if shipper_match else None

        consignee_match = re.search(r'2\)\s*([\s\S]+?)(?=\n\s*3\))', party_block)
        consignee = clean_text(consignee_match.group(1)) if consignee_match else None

        notify_match = re.search(r'3\)\s*([\s\S]+?)(?=\*\* TEL:|NOTIFY 2:|$)', party_block)
        notify = clean_text(notify_match.group(1)) if notify_match else None
        
        # Final destination is in the header of the page containing the B/L
        delivery_match = re.search(r'Final destination\s+([^\n]+)', content_block)
        place_of_delivery = clean_text(delivery_match.group(1)) if delivery_match else None

        # Extract from content_block
        weight_match = re.search(r'1\)\s*([\d,.]+)\s*KGS', content_block)
        weight = weight_match.group(1).replace(',', '') if weight_match else None
        if not weight:
            weight_match_alt = re.search(r'GROSS WEIGHT\s*[:=]?\s*([\d,.\s]+)\s*KGS', content_block, re.IGNORECASE)
            if weight_match_alt:
                weight = weight_match_alt.group(1).strip().replace(',', '').split('.')[0]

        volume_match = re.search(r'2\)\s*([\d,.]+)\s*CBM', content_block)
        volume = volume_match.group(1).replace(',', '') if volume_match else None
        if not volume:
            volume_match_alt = re.search(r'VOLUME IN CBM \(M3\)\s*[:=]?\s*([\d,.\s]+)\s*CBM', content_block, re.IGNORECASE)
            if volume_match_alt:
                volume = volume_match_alt.group(1).strip().replace(',', '')

        pack_qty_match = re.search(r'([\d,]+)\s+(BALE|BAG|CARTON)', content_block, re.IGNORECASE)
        pack_qty = pack_qty_match.group(1).replace(',', '') if pack_qty_match else None
        pack_unit = pack_qty_match.group(2).upper() + 'S' if pack_qty_match and not pack_qty_match.group(2).upper().endswith('S') else (pack_qty_match.group(2).upper() if pack_qty_match else None)

        desc_match = re.search(r'CONTAINER\(S\)\s+SAID\s+TO\s+CONTAIN:\s*([\s\S]+?)(?=NET WEIGHT|GROSS WEIGHT|HS CODE|FREIGHT PAYABLE|DEC NO|D6 E|SERVICE CONTRACT)', content_block, re.IGNORECASE)
        description = clean_text(desc_match.group(1)) if desc_match else None

        container_matches = list(re.finditer(r'\(\s*CN\s*\)\s*([A-Z]{4}\d{7})', content_block))

        if not container_matches:

                'pack_unit': pack_unit,
                'description': description,
            })
            continue

            container_number = c_match.group(1)

            type_match = re.search(r'\(CY/CY\)\s*-\s*([A-Z0-9]+)', container_sub_block)
            container_type = type_match.group(1) if type_match else None

            seal_match = re.search(r'\(\s*SN\s*\)\s*([^\n]+)', container_sub_block)
            seal1 = None
            if seal_match:
                seals = re.split(r'[,/\s]+', seal_match.group(1).strip())
                seal1 = seals[0] if seals and seals[0] else None

            base_row = {
                'bl_number': bl_number,
                'shipper': shipper,
                'consignee': consignee,
                'notify': notify,

                'pack_unit': pack_unit,
                'description': description,
                'container_number': container_number,
                'container_type': container_type,
                'seal1': seal1,
            }
            rows.append(base_row)

    return rows
