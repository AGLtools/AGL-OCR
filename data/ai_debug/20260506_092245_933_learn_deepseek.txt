def parse(text: str) -> list:
    rows = []

    def clean_value(value):
        if not value:
            return None
        cleaned = re.sub(r'^\s*\d\)\s*', '', value)
        cleaned = cleaned.replace('**', '').strip()
        return ' '.join(cleaned.replace('\n', ' ').split())

    # ports from first page (document-level)
    pol_match = re.search(r'5a\.Port of loading\s+([^\n]+)', text)
    pol = clean_value(pol_match.group(1)) if pol_match else None

    pod_match = re.search(r'5b\.Port of discharge\s+([^\n]+)', text)
    pod = clean_value(pod_match.group(1)) if pod_match else None

    # find all BL numbers
    bl_matches = list(re.finditer(r'\b(ABJ\d{9})\b', text))
    if not bl_matches:
        return rows

    # vessel/voyage/date static
    vessel = "KOTA SINGA"
    voyage = "KSNG0079E"
    sailing_date = "2026-03-21"

    for i, bl_match in enumerate(bl_matches):
        bl_number = bl_match.group(1)
        # block for this BL: from this BL number to the next BL number (or end)
        start = bl_match.end()
        end = bl_matches[i+1].start() if i+1 < len(bl_matches) else len(text)
        block = text[start:end]

        # ---- EXTRACT PARTIES (shipper, consignee, notify) ----
        shipper = consignee = notify = None

        # shipper: between "1)" and "2)"
        shp_match = re.search(r'\b1\)\s*([\s\S]+?)\n\s*2\)', block)
        if shp_match:
            shipper = clean_value(shp_match.group(1))
        else:
            shp_match = re.search(r'\b1\)\s*([\s\S]+?)(?=\n\s*(?:2\)|B/L\.))', block)
            if shp_match:
                shipper = clean_value(shp_match.group(1))

        # consignee: between "2)" and "3)"
        cee_match = re.search(r'\b2\)\s*([\s\S]+?)\n\s*3\)', block)
        if cee_match:
            consignee = clean_value(cee_match.group(1))
        else:
            cee_match = re.search(r'\b2\)\s*([\s\S]+?)(?=\n\s*(?:3\)|B/L\.|ABJ\d{9}|\*\* TEL:|NOTIFY|Final destination))', block)
            if cee_match:
                consignee = clean_value(cee_match.group(1))

        # notify: between "3)" and next marker
        nfy_match = re.search(r'\b3\)\s*([\s\S]+?)(?=\n\s*(?:B/L\. NO\.|ABJ\d{9}|\*\* TEL:|NOTIFY 2:|Final destination))', block)
        if nfy_match:
            notify = clean_value(nfy_match.group(1))

        # ---- PLACE OF DELIVERY ----
        delivery_match = re.search(r'Final destination\s+([^\n]+)', block)
        place_of_delivery = clean_value(delivery_match.group(1)) if delivery_match else None

        # ---- WEIGHT (TOTAL per BL) ----
        weight = None
        # Try to find "1) ... KGS" pattern after description
        weight_match = re.search(r'(?:\d\)\s*)?(?:1\)|Gross WGT|TOTAL GROSS WEIGHT)\s*[:=]?\s*([\d,]+\.?\d*)\s*KGS', block, re.IGNORECASE)
        if weight_match:
            weight = weight_match.group(1).replace(',', '')
        else:
            # Fallback: look for line starting with "1)" followed by number KGS
            weight_match = re.search(r'\b1\)\s*([\d,]+\.?\d*)\s*KGS', block)
            if weight_match:
                weight = weight_match.group(1).replace(',', '')

        # ---- VOLUME (TOTAL per BL) ----
        volume = None
        vol_match = re.search(r'(?:\d\)\s*)?(?:2\)|MEAS|VOLUME)\s*[:=]?\s*([\d,]+\.?\d*)\s*CBM', block, re.IGNORECASE)
        if vol_match:
            volume = vol_match.group(1).replace(',', '')
        else:
            vol_match = re.search(r'\b2\)\s*([\d,]+\.?\d*)\s*CBM', block)
            if vol_match:
                volume = vol_match.group(1).replace(',', '')

                seal1 = seals[0] if seals and seals[0] else None

            row = {
                'bl_number': bl_number,
                'container_number': container_number,
                'container_type': container_type,
                'seal1': seal1,
                'weight': weight,
                'volume': volume,

                'weight_unit': 'KGS' if weight else None,
                'volume_unit': 'CBM' if volume else None,
                'bl_type': 'MBL',
            }
            rows.append(row)

    return rows
