import re

def parse(text: str) -> list:
    rows = []
    
    # Regex pour extraire les informations du BL
    bl_pattern = re.compile(
        r"SH:\s*(?P<shipper>.*?)\s*CO:\s*(?P<consignee>.*?)\s*NF:\s*(?P<notify>.*?)(?=\s*EW:|\s*Total No\. of containers:)",
        re.DOTALL
    )
    
    # Regex pour extraire les informations du BL Number, Port of Loading, Port of Discharge
    bl_header_pattern = re.compile(
        r"B/L NR\.\s*(?P<bl_number>MEDU[A-Z0-9]+)\s*.*?"
        r"5a\. PORT OF LOADING\s*(?P<port_of_loading>.*?)\s*5b\. PORT OF DISCHARGE\s*(?P<port_of_discharge>.*?)\s*FINAL DESTINATION",
        re.DOTALL
    )

    # Regex pour extraire les informations de conteneur
    container_pattern = re.compile(
        r"CN:(?P<container_number>[A-Z]{4}\d{7})\s*(?P<container_type>[0-9]{2}'\s*[A-Z\s]+?)(?:\s+EXPRESS RELEASE)?\s*SN:\s*\d+\s*Tare Weight:\s*[\d,\.]+\s*kgs\.\s*"
        r"(?P<pack_qty>\d+)\s*(?P<pack_unit>[A-Z\s]+?)\s*of\s*(?P<description>.*?)\s*(?P<weight>[\d,\.]+)(?P<weight_unit>kgs\.|lbs\.)",
        re.DOTALL
    )

    # Regex pour extraire le numéro de page
    page_pattern = re.compile(r"Page (\d+) of \d+")

    current_page = "1"
    
    # Split the text into potential BL blocks
    # A new BL block starts with "SH:" or "MEDITERRANEAN DEPARTURE DATE Page" (for new page with header)
    bl_blocks = re.split(r"(?=(?:SH:|MEDITERRANEAN DEPARTURE DATE Page))", text)

    for block in bl_blocks:
        if not block.strip():
            continue

        # Update page number
        page_match = page_pattern.search(block)
        if page_match:
            current_page = page_match.group(1)

        bl_data = {}
        
        # Try to extract BL header info (bl_number, ports)
        bl_header_match = bl_header_pattern.search(block)
        if bl_header_match:
            bl_data['bl_number'] = bl_header_match.group('bl_number').strip()
            bl_data['port_of_loading'] = bl_header_match.group('port_of_loading').strip()
            bl_data['port_of_discharge'] = bl_header_match.group('port_of_discharge').strip()
        
        # Try to extract SH, CO, NF
        bl_info_match = bl_pattern.search(block)
        if bl_info_match:
            bl_data['shipper'] = re.sub(r'\s+', ' ', bl_info_match.group('shipper')).strip()
            bl_data['consignee'] = re.sub(r'\s+', ' ', bl_info_match.group('consignee')).strip()
            bl_data['notify'] = re.sub(r'\s+', ' ', bl_info_match.group('notify')).strip()
        
        # Extract container information within the current block
        container_matches = container_pattern.finditer(block)
        
        for match in container_matches:
            row = bl_data.copy() # Start with common BL data
            row['container_number'] = match.group('container_number').strip()
            row['container_type'] = match.group('container_type').strip()
            row['pack_qty'] = int(float(match.group('pack_qty').replace(',', '')))
            row['pack_unit'] = match.group('pack_unit').strip()
            row['description'] = re.sub(r'\s+', ' ', match.group('description')).strip()
            row['weight'] = float(match.group('weight').replace(',', ''))
            row['weight_unit'] = match.group('weight_unit').strip().replace('.', '')
            row['page'] = current_page
            rows.append(row)

    return rows
<<<END_PYTHON>>>
