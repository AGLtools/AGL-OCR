def parse(text: str) -> list:
    rows = []

    content_match = re.search(r'=== PAGE 2 ===(.*)', text, re.DOTALL)
    if not content_match:
        return rows
    content = content_match.group(1)

    bl_data = {}
    potential_blocks = re.split(r'(?=\n1\)\s[A-Z])', content)

    for p_block in potential_blocks:
        bl_match = re.search(r'\b(ABJ\d{8,10})\b', p_block)
        if bl_match:
            bl_num = bl_match.group(1)
            if bl_num not in bl_data:
                bl_data[bl_num] = ''
            bl_data[bl_num] += p_block

    for bl_number, block in bl_data.items():
        
        def clean_text(s):
            return re.sub(r'\s+', ' ', s).strip()

        shipper_match = re.search(r'1\)\s(.*?)(?=\n\s*2\))', block, re.DOTALL)
        shipper = clean_text(shipper_match.group(1)) if shipper_match else ''

        consignee_match = re.search(r'2\)\s(.*?)(?=\n\s*3\))', block, re.DOTALL)
        consignee = clean_text(consignee_match.group(1)) if consignee_match else ''

        notify_match = re.search(r'3\)\s(.*?)(?=\n\s*\(MN\)|\n\s*B/L\. NO\.|\n\s*\*\*|\n\s*NOTIFY 2|Final destination)', block, re.DOTALL)
        notify = clean_text(notify_match.group(1)) if notify_match else ''

        weight_match = re.search(r'1\)\s*([\d,]+\.?\d*)\s*KGS', block)
        weight = weight_match.group(1).replace(',', '') if weight_match else None

        volume_match = re.search(r'2\)\s*([\d,]+\.?\d*)\s*CBM', block)
        volume = volume_match.group(1).replace(',', '') if volume_match else None

        pack_match = re.search(r'([\d,]+)\s+([A-Z]+)\n.*?CONTAINER\(S\)', block)
        pack_qty = pack_match.group(1).replace(',', '') if pack_match else None
        pack_unit = pack_match.group(2) if pack_match else None
        
        desc_match = re.search(r'CONTAINER\(S\) SAID TO CONTAIN:\n(.*?)(?=\n\s*1\)\s*[\d,]+\.?\d*\s*KGS|\n\s*Colis:)', block, re.DOTALL)
        description = clean_text(desc_match.group(1)) if desc_match else ''

        common_data = {
            'bl_number': bl_number,
            'shipper': shipper,
            'consignee': consignee,
            'notify': notify,
            'description': description,
        }
        if weight:
            common_data['weight'] = weight
            common_data['weight_unit'] = 'KGS'
        if volume:
            common_data['volume'] = volume
            common_data['volume_unit'] = 'CBM'
        if pack_qty:
            common_data['pack_qty'] = pack_qty
        if pack_unit:
            common_data['pack_unit'] = pack_unit

        cn_matches = list(re.finditer(r'\(CN\)\s*([A-Z]{4}\d{7})', block))

        if not cn_matches:
            if common_data.get('shipper'):
                rows.append(common_data)
        else:
            for i, match in enumerate(cn_matches):
                row = common_data.copy()
                container_number = match.group(1)
                
                start = match.start()
                end = cn_matches[i+1].start() if i + 1 < len(cn_matches) else len(block)
                cn_sub_block = block[start:end]
                
                seal_match = re.search(r'\(SN\)\s*([^,\n]+)', cn_sub_block)
                seal = clean_text(seal_match.group(1)) if seal_match else ''
                
                type_match = re.search(r'\(CY/CY\)\s*-\s*([A-Z0-9]+)', cn_sub_block)
                container_type = type_match.group(1).strip() if type_match else ''
                
                row['container_number'] = container_number
                row['seal1'] = seal
                row['container_type'] = container_type
                
                rows.append(row)
                
    return rows
