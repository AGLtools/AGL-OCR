def parse(text: str) -> list:
    rows = []
    
    # Split the entire text into BL blocks. The delimiter is a new line starting with "SH: ".
    bl_texts = re.split(r'\nSH:\s+', text)
    if len(bl_texts) < 2:
        return []

    for bl_text in bl_texts[1:]:
        # Extract common BL info
        bl_number_match = re.search(r'^([A-Z0-9]+)', bl_text)
        bl_number = bl_number_match.group(1) if bl_number_match else None

        shipper_match = re.search(r'^[^\n]+?\n(.*?)(?=\nCO:)', bl_text, re.DOTALL)
        shipper = ' '.join(shipper_match.group(1).strip().split('\n')) if shipper_match else None

        consignee_match = re.search(r'\nCO:\n(.*?)(?=\nNF:)', bl_text, re.DOTALL)
        consignee = ' '.join(consignee_match.group(1).strip().split('\n')) if consignee_match else None

        notify_match = re.search(r'\nNF:\n(.*?)(?=\nEW:|\nTotal No. of containers:)', bl_text, re.DOTALL)
        notify = ' '.join(notify_match.group(1).strip().split('\n')) if notify_match else None

        # Split the BL block into container sub-blocks. The delimiter is a new line starting with "CN:".
        container_chunks = re.split(r'\n\s*CN:', bl_text)

        for chunk in container_chunks:
            if not chunk.strip():
                continue

            container_number_match = re.search(r'([A-Z]{4}\d{7})', chunk)
            if not container_number_match:
                continue

            container_number = container_number_match.group(1)
            
            container_type_match = re.search(r'[A-Z]{4}\d{7}\s+(.*?)(?:\n|$)', chunk)
            container_type = container_type_match.group(1).strip() if container_type_match else None

            seal_match = re.search(r'SN:\s*(\S+)', chunk)
            seal1 = seal_match.group(1) if seal_match else None

            # Isolate the block containing cargo details (description, packages, weight)
            start_index = 0
            if seal_match:
                start_index = seal_match.end()
            elif container_type_match:
                start_index = container_type_match.end()
            
            cargo_block = chunk[start_index:]
            
            # Clean up irrelevant info from the cargo block
            cargo_block = re.sub(r'Tare Weight:.*', '', cargo_block)
            cargo_block = re.sub(r'Harmonized Code:.*', '', cargo_block)

            # Extract weight
            weight_match = re.search(r'(\d[\d,.]*)\s*kgs', cargo_block)
            weight = weight_match.group(1).replace(',', '') if weight_match else None
            
            # Extract packages
            pack_match = re.search(r'(\d+)\s+([A-Z]+\(S\))', cargo_block)
            pack_qty = pack_match.group(1) if pack_match else None
            pack_unit = pack_match.group(2) if pack_match else None

            # The rest is description. We clean it by removing already extracted parts.
            desc_text = cargo_block
            if weight_match:
                desc_text = desc_text.replace(weight_match.group(0), '')
            if pack_match:
                desc_text = desc_text.replace(pack_match.group(0), '')
            
            desc_text = re.sub(r'\d[\d,.]*\s*lbs', '', desc_text)
            
            description = ' '.join(desc_text.strip().split())
            if pack_match:
                description = f"{pack_match.group(0)} {description}".strip()

            rows.append({
                'bl_number': bl_number,
                'shipper': shipper,
                'consignee': consignee,
                'notify': notify,
                'container_number': container_number,
                'container_type': container_type,
                'seal1': seal1,
                'weight': weight,
                'weight_unit': 'KGS' if weight else None,
                'pack_qty': pack_qty,
                'pack_unit': pack_unit,
                'description': description,
            })

    return rows
<<<END_PYTHON>>>
