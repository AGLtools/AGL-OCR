def parse(text: str) -> list:
    rows = []

    # Split the text by B/L number. The pattern keeps the delimiter.
    # Prepending a newline helps catch the first block correctly.
    blocks = re.split(r'\b(ABJ\d{9,11})\b', '\n' + text)
    
    # Iterate over pairs of (bl_number, content_block)
    for i in range(1, len(blocks), 2):
        bl_number = blocks[i]
        party_block = blocks[i-1]
        content_block = blocks[i+1]

        # --- Extract Shipper, Consignee, Notify from the block preceding the B/L number ---
        shipper = consignee = notify = None
        
        # Clean up the party block from common headers/footers to improve accuracy
        party_search_area = re.split(r'OUTWARD FOREIGN CARGO MANIFEST', party_block)[-1]
        party_search_area = re.split(r'SERVICE\s+MANIFESTE', party_search_area)[-1]

        shipper_re = re.search(r'1\)\s*(?!SHIPPER)([\s\S]+?)\s*2\)\s*', party_search_area, re.DOTALL)
        consignee_re = re.search(r'2\)\s*(?!CONSIGNEE)([\s\S]+?)\s*3\)\s*', party_search_area, re.DOTALL)
        notify_re = re.search(r'3\)\s*(?!NOTIFY)([\s\S]+?)(?=\*{2}|\Z)', party_search_area, re.DOTALL)

        if shipper_re:
            shipper = ' '.join(shipper_re.group(1).replace('\n', ' ').strip().split())
        if consignee_re:
            consignee = ' '.join(consignee_re.group(1).replace('\n', ' ').strip().split())
        if notify_re:
            notify = ' '.join(notify_re.group(1).replace('\n', ' ').strip().split())

        # --- Find all containers and their specific data ---
        container_matches = list(re.finditer(r'\(\s*CN\s*\)\s*([A-Z]{4}\d{7})', content_block))
        
        bl_rows = []
        has_per_container_weight = False

        if not container_matches:
            # Handle case with no containers (bulk cargo)
            bl_rows.append({
                'bl_number': bl_number,
                'shipper': shipper,
                'consignee': consignee,
                'notify': notify,
                'container_number': None,
            })
        else:
            for j, c_match in enumerate(container_matches):
                container_number = c_match.group(1)
                
                # Define a sub-block for this container to find its specific details
                start_pos = c_match.start()
                end_pos = container_matches[j+1].start() if j + 1 < len(container_matches) else len(content_block)
                sub_block = content_block[start_pos:end_pos]

                seal_match = re.search(r'\(\s*SN\s*\)\s*([^\n\s/,]+)', sub_block)
                seal = seal_match.group(1).strip() if seal_match else None
                
                # Look for per-container weight (at least 4 digits to be safe)
                weight = None
                weight_match = re.search(r'(\d{4,})\s*k?gs?', sub_block, re.IGNORECASE)
                if weight_match:
                    weight = weight_match.group(1)
                    has_per_container_weight = True

                bl_rows.append({
                    'bl_number': bl_number,
                    'shipper': shipper,
                    'consignee': consignee,
                    'notify': notify,
                    'container_number': container_number,
                    'seal1': seal,
                    'weight': weight,
                })

        # --- Extract BL-level data (totals, description) ---
        
        total_weight = None
        if not has_per_container_weight:
            weight_match = re.search(r'(?:TOTAL\s+GROSS\s+WEIGHT|GROSS\s+WEIGHT|1\))\s*[:=]?\s*([\d,]+\.?\d*)\s*KGS', content_block, re.IGNORECASE)
            if weight_match:
                total_weight = weight_match.group(1).replace(',', '').split('.')[0]
        
        volume_match = re.search(r'(?:VOLUME\s+IN\s+CBM|2\))\s*[:=]?\s*([\d,]+\.?\d*)\s*CBM', content_block, re.IGNORECASE)
        total_volume = volume_match.group(1).replace(',', '').split('.')[0] if volume_match else None

        pack_qty_match = re.search(r'([\d,.]+)\s+(BALE|BAG|CARTON)S?', content_block, re.IGNORECASE)
        pack_qty = pack_qty_match.group(1).replace(',', '').replace('.', '') if pack_qty_match else None
        pack_unit = pack_qty_match.group(2).upper() + 'S' if pack_qty_match and not pack_qty_match.group(2).upper().endswith('S') else (pack_qty_match.group(2).upper() if pack_qty_match else None)

        desc_match = re.search(r'CONTAINER\(S\)\s+SAID\s+TO\s+CONTAIN:\s*([\s\S]+?)(?=NET\s+WEIGHT|GROSS\s+WEIGH|HS\s+CODE|FREIGHT\s+PAYABLE|DEC\s+NO|SERVICE\s+CONTRACT)', content_block, re.IGNORECASE | re.DOTALL)
        description = ' '.join(desc_match.group(1).replace('\n', ' ').strip().split()) if desc_match else None

        # --- Apply the BL-level data to all rows of this BL ---
        for row in bl_rows:
            if not row.get('weight'): # Only apply total if per-container was not found
                row['weight'] = total_weight
            row['volume'] = total_volume
            row['pack_qty'] = pack_qty
            row['pack_unit'] = pack_unit
            row['description'] = description
        
        rows.extend(bl_rows)

    return rows
