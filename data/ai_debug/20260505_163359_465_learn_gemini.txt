def parse(text: str) -> list:
    rows = []
    
    # Split text into BL blocks. A block starts with "1) [Shipper Name]".
    # We use a regex that looks for "1) " not followed by a number, which distinguishes it from weight lines.
    bl_blocks = re.split(r'\n1\)\s+(?![0-9,\s]+(?:KGS|CBM))', text)

    for i, block in enumerate(bl_blocks):
        if i == 0:
            continue # Skip header part before the first BL
        
        full_block = "1) " + block

        bl_match = re.search(r'\b(ABJ\d{8,10})\b', full_block)
        if not bl_match:
            continue
        
        bl_number = bl_match.group(1)

        shipper_match = re.search(r'1\)\s*(.*?)(?=\n\s*2\))', full_block, re.DOTALL)
        consignee_match = re.search(r'2\)\s*(.*?)(?=\n\s*3\))', full_block, re.DOTALL)
        notify_match = re.search(r'3\)\s*(.*?)(?=\n\s*(?:\*\*|ABJ\d{8,10}|\(MN\)))', full_block, re.DOTALL)
        
        shipper = shipper_match.group(1).replace('\n', ' ').strip() if shipper_match else None
        consignee = consignee_match.group(1).replace('\n', ' ').strip() if consignee_match else None
        notify = notify_match.group(1).replace('\n', ' ').strip() if notify_match else None

        pod_match = re.search(r'5b\.Port of discharge\n\s*([^\n]+)', full_block)
        port_of_discharge = pod_match.group(1).strip() if pod_match else None
        
        delivery_match = re.search(r'Final destination\n\s*([^\n]+)', full_block)
        place_of_delivery = delivery_match.group(1).strip() if delivery_match else None

        pack_qty_match = re.search(r'([\d,]+)\s+(BALE|BAG|CARTON|BALES|BAGS|CARTONS)', full_block, re.IGNORECASE)
        pack_qty = None
        pack_unit = None
        if pack_qty_match:
            try:
                pack_qty = int(pack_qty_match.group(1).replace(',', ''))
                pack_unit = pack_qty_match.group(2).upper()
                if not pack_unit.endswith('S'):
                    pack_unit += 'S'
                if pack_unit == 'BALE':
                    pack_unit = 'BALES'
            except (ValueError, IndexError):
                pass

        desc_match = re.search(r'\d+X\d+\w*\s*CONTAINER\(S\)\s*SAID TO CONTAIN:\n(.*?)\n', full_block, re.DOTALL | re.IGNORECASE)
        description = desc_match.group(1).replace('\n', ' ').strip() if desc_match else None

        weight_match = re.search(r'1\)\s*([\d,.]+)\s*KGS', full_block)
        total_weight = float(weight_match.group(1).replace(',', '')) if weight_match else 0.0

        volume_match = re.search(r'2\)\s*([\d,.]+)\s*CBM', full_block)
        total_volume = float(volume_match.group(1).replace(',', '')) if volume_match else 0.0

        container_matches = list(re.finditer(r'\(CN\)\s*([A-Z]{4}\d{7})', full_block))
        num_containers = len(container_matches) if container_matches else 1
        
        weight_per_container = total_weight / num_containers if num_containers > 0 else 0.0
        volume_per_container = total_volume / num_containers if num_containers > 0 else 0.0

        bl_info = {
            'bl_number': bl_number,
            'shipper': shipper,
            'consignee': consignee,
            'notify': notify,
            'port_of_discharge': port_of_discharge,
            'place_of_delivery': place_of_delivery,
            'description': description,
            'weight_unit': 'KGS' if total_weight > 0 else None,
            'volume_unit': 'CBM' if total_volume > 0 else None,
        }
        if pack_qty:
            bl_info['pack_qty'] = pack_qty
        if pack_unit:
            bl_info['pack_unit'] = pack_unit

        if not container_matches:
            row = bl_info.copy()
            if total_weight > 0:
                row['weight'] = total_weight
            if total_volume > 0:
                row['volume'] = total_volume
            rows.append(row)
        else:
            for i, match in enumerate(container_matches):
                container_number = match.group(1)
                
                start_pos = match.start()
                end_pos = container_matches[i+1].start() if i + 1 < len(container_matches) else len(full_block)
                container_block = full_block[start_pos:end_pos]

                seal_match = re.search(r'\(SN\)\s*([^\n]+)', container_block)
                seal1 = seal_match.group(1).strip().split('/')[0].strip() if seal_match else None

                row = bl_info.copy()
                row['container_number'] = container_number
                row['seal1'] = seal1
                if weight_per_container > 0:
                    row['weight'] = weight_per_container
                if volume_per_container > 0:
                    row['volume'] = volume_per_container
                rows.append(row)

    return rows
