import re

def parse(text: str) -> list:
    rows = []
    
    # Regex pour extraire les informations d'en-tête (communes à tous les BL)
    vessel_match = re.search(r"NAME OF SHIP\s*\n\s*MSC\s*([A-Z0-9\s]+)\s*\(", text)
    vessel = vessel_match.group(1).strip() if vessel_match else None

    voyage_match = re.search(r"VOYAGE\s*\n\s*([A-Z0-9]+)", text)
    voyage = voyage_match.group(1).strip() if voyage_match else None

    date_of_arrival_match = re.search(r"Date Of Arrival at Port of Discharge\s*\n\s*(\d{2} \w{3} \d{4})", text)
    date_of_arrival = date_of_arrival_match.group(1).strip() if date_of_arrival_match else None

    # Regex pour découper le document en blocs de BL
    # Un bloc commence par "SH:" ou "B/L NR." et se termine avant le prochain "SH:" ou "B/L NR." ou "--- FIN ---"
    bl_blocks = re.split(r'(?=\nSH:|\nB/L NR\.)', text)

    current_page = 1
    page_match = re.search(r"Page (\d+) of \d+", text)
    if page_match:
        current_page = int(page_match.group(1))

    for block in bl_blocks:
        if not block.strip():
            continue

        # Extraire le numéro de page pour ce bloc
        page_num_match = re.search(r"Page (\d+) of \d+", block)
        if page_num_match:
            current_page = int(page_num_match.group(1))

        bl_data = {
            "vessel": vessel,
            "voyage": voyage,
            "date_of_arrival": date_of_arrival,
            "page": current_page
        }

        # Extraire le numéro de BL
        bl_number_match = re.search(r"B/L NR\.\s*(\w+)", block)
        if bl_number_match:
            bl_data["bl_number"] = bl_number_match.group(1).strip()
        else:
            # Si pas de B/L NR. explicite, chercher le numéro après SH:
            bl_number_match = re.search(r"SH:\s*(\w+)", block)
            if bl_number_match:
                bl_data["bl_number"] = bl_number_match.group(1).strip()
            else:
                continue # Skip blocks without a clear BL number

        # Extraire Shipper
        shipper_match = re.search(r"SH:\s*\w+\s*(.*?)(?=\nCO:|\nNF:|\nCN:|\nTotal No\. of containers:)", block, re.DOTALL)
        if shipper_match:
            shipper_lines = [line.strip() for line in shipper_match.group(1).split('\n') if line.strip()]
            bl_data["shipper"] = "\n".join(shipper_lines).strip()

        # Extraire Consignee
        consignee_match = re.search(r"CO:\s*(.*?)(?=\nNF:|\nCN:|\nTotal No\. of containers:)", block, re.DOTALL)
        if consignee_match:
            consignee_lines = [line.strip() for line in consignee_match.group(1).split('\n') if line.strip()]
            bl_data["consignee"] = "\n".join(consignee_lines).strip()

        # Extraire Notify
        notify_match = re.search(r"NF:\s*(.*?)(?=\nCN:|\nTotal No\. of containers:)", block, re.DOTALL)
        if notify_match:
            notify_lines = [line.strip() for line in notify_match.group(1).split('\n') if line.strip()]
            bl_data["notify"] = "\n".join(notify_lines).strip()

        # Extraire Port of Loading
        pol_match = re.search(r"5a\. PORT OF LOADING\s*\n\s*(.*?)(?=\s*5b\. PORT OF DISCHARGE)", block, re.DOTALL)
        if pol_match:
            bl_data["port_of_loading"] = pol_match.group(1).strip()

        # Extraire Port of Discharge
        pod_match = re.search(r"5b\. PORT OF DISCHARGE\s*\n\s*(.*?)(?=\s*FINAL DESTINATION)", block, re.DOTALL)
        if pod_match:
            bl_data["port_of_discharge"] = pod_match.group(1).strip()

        # Extraire Place of Delivery (Final Destination)
        podel_match = re.search(r"FINAL DESTINATION\s*(.*?)(?=\s*Transshipment:)", block, re.DOTALL)
        if podel_match:
            bl_data["place_of_delivery"] = podel_match.group(1).strip()

        # Extraire les conteneurs
        container_blocks = re.findall(r"(CN:([A-Z]{4}\d{7})\s*(.*?)(?=(?:CN:|\nTotal No\. of containers:|\nEW:|\Z)))", block, re.DOTALL)
        
        if not container_blocks: # Cas où il n'y a qu'un seul conteneur ou pas de CN: explicite
            # Tenter d'extraire les infos de description, poids, colis pour le BL entier

            description = description_match.group(1).strip() if description_match else None

            weight_match = re.search(r"GROSS WEIGHT:\s*([\d\.,]+)\s*(KGS|LBS)", block, re.IGNORECASE)
            weight = float(weight_match.group(1).replace(',', '')) if weight_match else None
            weight_unit = weight_match.group(2).upper() if weight_match else None

            pack_qty_match = re.search(r"(\d+)\s*(PALLET\(S\)|BAG\(S\))", block)
            pack_qty = int(pack_qty_match.group(1)) if pack_qty_match else None
            pack_unit = pack_qty_match.group(2).replace('(S)', '') if pack_qty_match else None

            if description or weight or pack_qty:
                row = bl_data.copy()
                row["description"] = description
                row["weight"] = weight
                row["weight_unit"] = weight_unit
                row["pack_qty"] = pack_qty
                row["pack_unit"] = pack_unit
                rows.append(row)
            continue

            if container_type_match:
                row["container_type"] = container_type_match.group(1).strip()

            # Description des marchandises
            description_match = re.search(r"(?:\d+\s*(?:PALLET\(S\)|BAG\(S\))\s*of\s*)(.*?)(?=\n\s*\d+\s*kgs\.|\n\s*\d+\s*lbs\.|\n\s*Harmonized Code:|\n\s*Tare Weight:|\Z)", container_details, re.DOTALL)
            if description_match:
                row["description"] = description_match.group(1).strip()
            else: # Fallback for description if not found with the above pattern
                description_match_alt = re.search(r"DESCRIPTION OF GOODS\s*(.*?)(?=\n\s*\d+\s*-GROSS WEIGHT)", block, re.DOTALL)
                if description_match_alt:
                    row["description"] = description_match_alt.group(1).strip()


            # Poids brut
            weight_match = re.search(r"([\d\.,]+)\s*(kgs|lbs)\.", container_details, re.IGNORECASE)
            if weight_match:
                row["weight"] = float(weight_match.group(1).replace(',', ''))
                row["weight_unit"] = weight_match.group(2).upper()

            # Quantité et unité de colis
            pack_qty_match = re.search(r"(\d+)\s*(PALLET\(S\)|BAG\(S\))", container_details)
            if pack_qty_match:
                row["pack_qty"] = int(pack_qty_match.group(1))

            rows.append(row)

    return rows
