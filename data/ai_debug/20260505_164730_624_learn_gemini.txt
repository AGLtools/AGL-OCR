    def clean_text(s):

            return ''
        return re.sub(r'\s+', ' ', s).strip()

    parts = re.split(bl_pattern, text)

    if len(parts) < 3:
        return []

    for i in range(1, len(parts), 2):
        bl_number = parts[i]
        content_block = parts[i+1]
        party_block = parts[i-1]

        shipper_match = re.search(r'1\)\s*(?:SHIPPER\s*)?(.*?)(?=\n\s*2\))', party_block, re.DOTALL | re.IGNORECASE)
