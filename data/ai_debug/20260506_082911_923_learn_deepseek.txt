def parse(text: str) -> list:
    rows = []
    # Pattern pour les numéros de BL (PIL: ABJ + 9 chiffres)
    bl_pattern = r'ABJ\d{9}'
    # Trouver tous les starts de BL dans l'ordre
    bl_starts = [m.start() for m in re.finditer(bl_pattern, text)]
    if not bl_starts:
        return rows

    # Découper le texte en blocs basés sur les positions des BL
    for i, start in enumerate(bl_starts):
        bl_number = text[start:start+12]  # ABJ + 9 digits = 12 chars
        # Fin du bloc : début du prochain BL ou fin du texte
        if i + 1 < len(bl_starts):
            end = bl_starts[i+1]
        else:
            end = len(text)
        block = text[start:end]

        # Extraire tous les conteneurs (CN) dans ce bloc
        cnt_pattern = r'\(\s*CN\s*\)\s*([A-Z]{4}\d{7})'
        containers = re.findall(cnt_pattern, block)
        if not containers:
            continue  # pas de conteneur, ignorer (peut être un en-tête)

        # Extraire le poids total (ligne "1) XXX.XXX KGS")
        weight = None
        w_match = re.search(r'1\)\s*([\d,]+\.\d{3})\s*KGS', block)
        if w_match:
            try:
                weight = float(w_match.group(1).replace(',', ''))
            except ValueError:
                pass

        # Extraire le volume total (ligne "2) XXX.XXX CBM")
        volume = None
        v_match = re.search(r'2\)\s*([\d,]+\.\d{3})\s*CBM', block)
        if v_match:
            try:
                volume = float(v_match.group(1).replace(',', ''))
            except ValueError:
                pass

        # Créer une ligne par conteneur avec les mêmes poids/volume
        for cnt in containers:
            rows.append({
                'bl_number': bl_number,
                'container_number': cnt,
                'weight': weight,
                'volume': volume,
                'bl_type': '', # optionnel
                'container_type': '',
                'weight_unit': 'KGS' if weight is not None else '',
                'volume_unit': 'CBM' if volume is not None else '',
                'shipper': '',
                'consignee': '',
                'notify': '',
                'freight_forwarder': '',
                'port_of_loading': '',
                'port_of_discharge': '',
                'place_of_delivery': '',
                'place_of_acceptance': '',
                'description': '',
                'pack_qty': None,
                'pack_unit': '',
                'seal1': '',
                'seal2': '',
                'seal3': '',
                'page': ''
            })

    return rows
